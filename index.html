/* --- 手術位置 A：替換原有的 openScripture 函式 --- */ function openScripture(bookKey, chap, sec, endSec) { const path = SCRIPTURE_PHYSICAL_MAP[bookKey]; // 1. 計算主要來源 (LDS) 與 錨點位移邏輯 let primaryUrl = ""; if (path) { let idParam = sec ? `p${sec}` : ""; if (sec && endSec) idParam = `p${sec}-p${endSec}`; primaryUrl = `https://www.churchofjesuschrist.org/study/scriptures/${path}/${chap}?lang=zho`; if (sec) { // [手術刀] 錨點偏移：目標 > 1 則減 1，避免頂端遮蔽；若為 1 則維持 1 const anchor = parseInt(sec) > 1 ? parseInt(sec) - 1 : 1; primaryUrl += `&id=${idParam}#p${anchor}`; } } // 2. 計算備用連結 (信望愛) const backupUrl = `https://bible.fhl.net/new/read.php?chineses=${encodeURIComponent(bookKey)}&chap=${chap}`; const modal = document.getElementById('scripture-modal'); const title = document.getElementById('modal-title'); // 3. 植入雙來源按鈕與新分頁選項 (UI 佈局優化) title.innerHTML = `
${bookKey} 第 ${chap} 章
`; // 執行初始切換 const defaultUrl = primaryUrl || backupUrl; switchSrc(defaultUrl, !!primaryUrl); modal.style.display = 'block'; } /* --- 手術位置 B：在 openScripture 後方新增 switchSrc 工具函式 --- */ function switchSrc(url, isPrimary) { if (!url || url === "null") return; const frame = document.getElementById('scripture-frame'); const btnP = document.getElementById('btn-primary'); const btnB = document.getElementById('btn-backup'); frame.src = url; // 透過透明度提供視覺反饋 if(btnP && btnB) { btnP.style.opacity = isPrimary ? "1" : "0.4"; btnB.style.opacity = isPrimary ? "0.4" : "1"; } }
