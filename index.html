/* --- 手術部位：精確替換 index.html 中的 openScripture 函式 --- */ function openScripture(bookKey, chap, sec, endSec) { const path = SCRIPTURE_PHYSICAL_MAP[bookKey]; // 1. 計算主要來源 (LDS) 並處理錨點偏移 let primaryUrl = ""; if (path) { let idParam = sec ? `p${sec}` : ""; if (sec && endSec) idParam = `p${sec}-p${endSec}`; primaryUrl = `https://www.churchofjesuschrist.org/study/scriptures/${path}/${chap}?lang=zho`; if (sec) { // [手術位置] 錨點偏移：若 > 1 則減 1，避免頂端遮蔽；若為 1 則維持 1 const anchor = parseInt(sec) > 1 ? parseInt(sec) - 1 : 1; primaryUrl += `&id=${idParam}#p${anchor}`; } } // 2. 計算備用連結 (信望愛) const backupUrl = `https://bible.fhl.net/new/read.php?chineses=${encodeURIComponent(bookKey)}&chap=${chap}`; const modal = document.getElementById('scripture-modal'); const title = document.getElementById('modal-title'); // 3. 僅修改 title 內部的 innerHTML，不變動 modal 原有結構 title.innerHTML = `
${bookKey} 第 ${chap} 章
`; // 執行初始載入邏輯 const defaultUrl = primaryUrl || backupUrl; switchSrc(defaultUrl, !!primaryUrl); modal.style.display = 'block'; } /* --- 手術部位：在 closeScripture 之前植入 switchSrc 工具函式 --- */ function switchSrc(url, isPrimary) { if (!url || url === "null") return; const frame = document.getElementById('scripture-frame'); const btnP = document.getElementById('btn-primary'); const btnB = document.getElementById('btn-backup'); frame.src = url; // 透過透明度提供視覺反饋，不改動原有按鈕屬性 if(btnP && btnB) { btnP.style.opacity = isPrimary ? "1" : "0.4"; btnB.style.opacity = isPrimary ? "0.4" : "1"; } }
